\section{Introduction}
This PDF is the language rulebook for \lang{}.
It defines what our compiler will support across assignments A2--A4 (lexer, parser, and semantic analysis).
% For Assignment A2, the main deliverable is the lexical analyzer; this PDF is our language-specification write-up for the same.
The intent is to keep the language close to a small, C-like core so that it remains implementable using \texttt{lex} and \texttt{yacc} in C.

\paragraph{Reference baseline.}
We used the provided \emph{YAPL} language specification as a starting reference.
\lang{} follows the same overall C-like style (expressions, statements, declarations), but the set of supported constructs is chosen by us and includes a few small extensions listed in Section~\ref{sec:extensions}.

\section{Lexical specification}
\subsection{Whitespace}
Spaces, tabs, and newlines may appear between tokens.
They are ignored by the lexer, except that newlines end single-line comments.

\subsection{Comments}
\begin{itemize}
	\item \textbf{Single-line (C++ style):} \texttt{//} continues up to end of line.
	\item \textbf{Single-line (extension):} \texttt{\#} continues up to end of line.
	\item \textbf{Multi-line:} \texttt{/*} begins a comment that ends at the next \texttt{*/}.
\end{itemize}

If the input ends before a multi-line comment is closed, the lexer must report an ill-formed (unclosed) comment.

\subsection{Identifiers}
Identifiers follow the usual C rule.
\begin{lstlisting}
[A-Za-z_][A-Za-z0-9_]*
\end{lstlisting}

\subsection{Keywords}
The following words are reserved and cannot be used as identifiers.

\paragraph{Core keywords (C-like).}
\begin{center}
\begin{tabular}{llll}
\toprule
	\texttt{int} & \texttt{long} & \texttt{short} & \texttt{char} \\
	\texttt{float} & \texttt{double} & \texttt{void} & \texttt{struct} \\
	\texttt{extern} & \texttt{return} & \texttt{break} & \texttt{continue} \\
	\texttt{if} & \texttt{else} & \texttt{switch} & \texttt{case} \\
	\texttt{default} & \texttt{for} & \texttt{while} & \texttt{do} \\
\bottomrule
\end{tabular}
\end{center}

\paragraph{Keywords added in \lang{}.}
\begin{center}
\begin{tabular}{llll}
\toprule
	\texttt{elif} & \texttt{pass} & \texttt{try} & \texttt{except} \\
	\texttt{print} & \texttt{range} & \texttt{in} & \texttt{foreach} \\
\bottomrule
\end{tabular}
\end{center}

Any standard C keyword not listed above is treated as a normal identifier in \lang{}.

\subsection{Literals}
\paragraph{Integer literals.}
We support decimal integers and hexadecimal integers of the form \texttt{0x...}.
\begin{itemize}
	\item decimal: \texttt{0} or \texttt{[1-9][0-9]*}
	\item hexadecimal: \texttt{0x[0-9a-fA-F]+}
\end{itemize}

\paragraph{Floating-point literals.}
We support decimal floats such as \texttt{3.14}, \texttt{0.5}, and \texttt{10.}.

\paragraph{Character literals.}
Single-quoted characters like \texttt{'a'} are supported.
Basic escape sequences may be supported (e.g., \texttt{\textbackslash n}, \texttt{\textbackslash t}, \texttt{\textbackslash '}, \texttt{\textbackslash \textbackslash}).

\paragraph{String literals.}
Double-quoted strings like \texttt{"hello"} are supported.
Adjacent strings are concatenated (C-style): \texttt{"a" "b"}.

\subsection{Punctuation}
The following punctuation tokens are used in the grammar: \texttt{( )}, \texttt{[ ]}, \texttt{\{ \}}, \texttt{,}, \texttt{;}, \texttt{:}.

\section{Data types}
\subsection{Primitive types}
\lang{} supports the following built-in types:
\begin{itemize}
	\item \texttt{int}, \texttt{long}, \texttt{short}, \texttt{char}
	\item \texttt{float}, \texttt{double}
	\item \texttt{void}
\end{itemize}

\subsection{Derived types}
\begin{itemize}
	\item \textbf{Pointers:} any type may be pointer-qualified, e.g., \texttt{int *****p;}.
	\item \textbf{Arrays:} any non-function type may be an array type, e.g., \texttt{int a[10];}.
	\item \textbf{Structs:} user-defined aggregate types via \texttt{struct}.
	\item \textbf{Functions:} functions may return any non-array type (including pointers).
\end{itemize}

\section{Operators and precedence}
\label{sec:ops}
Unless stated otherwise, operator precedence and associativity follow C for the supported operators.

\paragraph{Special operators.}
\texttt{sizeof} is treated as an operator token (not a reserved keyword).
Supported forms are \texttt{sizeof(type)} and \texttt{sizeof(unary\_expr)}.
Type casts of the form \texttt{(type)expr} are supported.

\begin{table}[ht]
\centering
\begin{tabular}{ll}
\toprule
Category & Tokens \\
\midrule
Arithmetic & \texttt{+} \texttt{-} \texttt{*} \texttt{/} \texttt{\%} \\
Relational & \texttt{<} \texttt{>} \texttt{<=} \texttt{>=} \texttt{==} \texttt{!=} \texttt{<=>} \\
Logical & \texttt{\&\&} \texttt{||} \texttt{!} \\
Bitwise (subset) & \texttt{\&} \texttt{|} \texttt{\string^} \texttt{\~} \\
Assignment & \texttt{=} \\
Inc/Dec & \texttt{++} \texttt{--} \\
Pointer / address & \texttt{*} \texttt{\&} \\
Field access & \texttt{.} \texttt{->} \\
Calls / indexing & \texttt{( )} \texttt{[ ]} \\
Special & \texttt{sizeof} \\
\bottomrule
\end{tabular}
\caption{Operators in \lang{}}
\label{tab:ops}
\end{table}

\paragraph{Not supported.}
We do not support preprocessors/directives, ternary \texttt{?:}, compound assignments (\texttt{+=}, \texttt{-=}, \dots), shift operators (\texttt{<<}, \texttt{>>}), or varargs/ellipsis (\texttt{...}).

\section{Language constructs (overview)}
\subsection*{Declarations and definitions}
\begin{itemize}
	\item Global variables (including \texttt{extern} declarations).
	\item Function declarations (prototypes) and function definitions (with bodies), including recursion.
	\item Local declarations inside blocks; multiple declarators separated by commas.
	\item \texttt{struct} types and field access using \texttt{.} and \texttt{->}.
\end{itemize}

\subsection*{Statements}
\begin{itemize}
	\item Selection: \texttt{if} / \texttt{else} / \texttt{elif} ladders; \texttt{switch} / \texttt{case} with optional \texttt{default} and fall-through.
	\item Iteration: \texttt{for}, \texttt{while}, \texttt{do-while}, restricted \texttt{for (i in range(l, r))}, and \texttt{foreach}.
	\item Jump: \texttt{break}, \texttt{continue}, \texttt{return}.
	\item Empty/no-op: \texttt{pass;} and the empty statement \texttt{;}.
\end{itemize}

\section{Extensions over the YAPL baseline}
\label{sec:extensions}
The following are additions that are specific to \lang{}.

\begin{itemize}
	\item \textbf{\texttt{elif}}: Supported as syntactic sugar for \texttt{else if}. It can appear zero or more times between an \texttt{if} and a final optional \texttt{else}, keeping if-ladders readable without forcing deeper nesting.

	\item \textbf{\texttt{pass;}}: An explicit no-op statement. It is equivalent to writing an empty statement \texttt{;}, but it makes intent clear in places like empty loop bodies or placeholder branches.

	\item \textbf{\texttt{print(...)}:} The call syntax is the usual function-call form with parentheses, and we allow a comma-separated argument list (including a single argument).

	\item \textbf{\texttt{range(l, r)}:} A built-in that represents the integer sequence from \texttt{l} (inclusive) to \texttt{r} (exclusive). In \lang{}, it is primarily used by the restricted range-loop form.

	\item \textbf{Restricted range loop:} A restricted Python-style loop form:
	\begin{lstlisting}
for (i in range(l, r)) stmt
	\end{lstlisting}
	Restrictions: the loop variable must be an identifier, and the iterator must be exactly \texttt{range(l, r)} with two expressions. Informally it behaves like iterating \texttt{i = l} while \texttt{i < r}.

	\item \textbf{\texttt{\#} comments:} In addition to \texttt{//}, \texttt{\#} starts a single-line comment. Since preprocessors are not part of \lang{}, lines starting with \texttt{\#} are always treated as comments and not as directives.

	\item \textbf{Try/except:} Python-like exception handling using block syntax:
	\begin{lstlisting}
try { ... } 
except { ... }
	\end{lstlisting}
	Only a single catch-all \texttt{except} is supported. We do not require exception types, matching, or raising in the current version.

	\item \textbf{Foreach loop:} \texttt{foreach} iterates over arrays and over string literals. Each iteration assigns the current element to the loop variable.

	\textbf{Foreach syntax:}
	\begin{lstlisting}
foreach(x in array) { ... }
foreach(c in "hello") { ... }
	\end{lstlisting}

	\item \textbf{Optional operators (only if integration stays clean):} These will be included only if precedence/associativity handling in \texttt{yacc} remains clean.
	\begin{itemize}
		\item \texttt{**} exponent operator (intended right-associative and higher precedence than \texttt{* / \%}).
		\item \texttt{??} null-coalescing operator (intended right-associative).
	\end{itemize}
\end{itemize}

\section{Grammar / syntax rules}
This is a compact, lexer-friendly grammar sketch.
It is not meant to be a complete C grammar, but it defines the constructs we plan to accept.

\begin{lstlisting}
program        := external_decl*

external_decl  := decl ';'
                | func_proto ';'
                | func_def

func_proto     := type ptr* id '(' param_list? ')'
func_def       := type ptr* id '(' param_list? ')' compound_stmt

decl           := type decl_item (',' decl_item)*
decl_item      := ptr* id array_suffix* ('=' expr)?
array_suffix   := '[' expr? ']'

type           := prim_type | 'struct' id
prim_type      := 'int' | 'long' | 'short' | 'char' | 'float' | 'double' | 'void'
ptr            := '*'

param_list     := param (',' param)*
param          := type ptr* id array_suffix*

compound_stmt  := '{' decl_or_stmt* '}'
decl_or_stmt   := decl ';' | stmt

stmt           := compound_stmt
                | expr ';'
                | ';'
                | 'pass' ';'
                | if_stmt
                | while_stmt
                | do_while_stmt
                | for_stmt
                | range_for
                | foreach_stmt
                | switch_stmt
                | try_except
                | jump_stmt

if_stmt        := 'if' '(' expr ')' stmt
				 ( 'elif' '(' expr ')' stmt )*
				 ( 'else' stmt )?

while_stmt     := 'while' '(' expr ')' stmt
do_while_stmt  := 'do' stmt 'while' '(' expr ')' ';'
for_stmt       := 'for' '(' expr? ';' expr? ';' expr? ')' stmt
range_for      := 'for' '(' id 'in' 'range' '(' expr ',' expr ')' ')' stmt

foreach_stmt   := 'foreach' '(' id 'in' expr ')' compound_stmt

switch_stmt    := 'switch' '(' expr ')' '{'
				 ( 'case' const ':' stmt* )*
				 ( 'default' ':' stmt* )?
				 '}'

try_except     := 'try' compound_stmt 'except' compound_stmt

jump_stmt      := 'break' ';'
                | 'continue' ';'
                | 'return' expr? ';'

const          := int_lit | char_lit
\end{lstlisting}

\section{Semantic rules (planned for A4)}
This section states the semantic rules we intend to enforce in the semantic analysis phase.
For A2 (lexer) and A3 (parser), these are documented here but do not need to be fully enforced.

\subsection*{Scoping}
\begin{itemize}
	\item Block scoping: each \texttt{\{...\}} introduces a new scope.
	\item Shadowing is allowed: an inner declaration may reuse an outer name.
	\item Function parameters are in the function's outermost local scope.
\end{itemize}

\subsection*{Declarations and usage}
\begin{itemize}
	\item Variables and functions must be declared before use within the same scope (C-like rule).
	\item A function prototype and a function definition may both appear; the definition must match the declared signature.
\end{itemize}

\subsection*{Type checking (high level)}
\begin{itemize}
	\item Arithmetic operators apply to numeric types; relational operators produce an integer truth value.
	\item Assignment requires compatible types (with standard numeric promotions).
	\item Pointer operations: \texttt{\&} yields a pointer, \texttt{*} dereferences a pointer.
	\item Field access: \texttt{.} requires a struct value, \texttt{->} requires a pointer-to-struct.
	\item \texttt{sizeof} returns an integer type.
\end{itemize}

\subsection*{Foreach typing}
\begin{itemize}
	\item For \texttt{foreach(x in array)}, the loop variable \texttt{x} must be assignable from the element type of \texttt{array}.
	\item For \texttt{foreach(c in "...\")}, \texttt{c} is treated as a \texttt{char}.
\end{itemize}

\section{Example programs}
The following examples are small and designed to exercise the supported tokens and constructs.

\subsection*{Example 1: if/elif/else and print}
\begin{lstlisting}
int main() {
	int x = 5;
	if (x < 0) {
		print("neg");
	} elif (x == 0) {
		print("zero");
	} else {
		print("pos");
	}
	return 0;
}
\end{lstlisting}

\subsection*{Example 2: range-based loop}
\begin{lstlisting}
int main() {
	int i;
	for (i in range(0, 5)) {
		print(i);
	}
	return 0;
}
\end{lstlisting}

\subsection*{Example 3: foreach over an array and a string}
\begin{lstlisting}
int main() {
	int a[3];
	a[0] = 10; a[1] = 20; a[2] = 30;

	int x;
	foreach(x in a) {
		print(x);
	}

	char c;
	foreach(c in "hi") {
		print(c);
	}
	return 0;
}
\end{lstlisting}

\subsection*{Example 4: try/except and pass}
\begin{lstlisting}
int main() {
	try {
		pass;
		print("inside try");
	} except {
		print("inside except");
	}
	return 0;
}
\end{lstlisting}

\section{Restrictions and design choices}
\begin{itemize}
	% \item No preprocessor support (no \texttt{\#include}, \texttt{\#define}). The \texttt{\#} token starts a comment.
	% \item No ternary operator \texttt{?:}, no shift operators, and no compound assignments.
	\item Statements such as expression statements and jumps end with \texttt{;}. Compound statements use braces.
	\item We aim to keep the grammar small enough to be implementable cleanly in \texttt{lex}/\texttt{yacc}.
\end{itemize}
